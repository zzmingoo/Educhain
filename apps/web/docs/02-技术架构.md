### 3. 服务层设计

**API 基础封装**

```typescript
// src/services/api.ts
async function fetchWithAuth<T>(
  endpoint: string,
  options: RequestInit = {}
): Promise<ApiResponse<T>> {
  const url = `${API_BASE_URL}${endpoint}`;
  const accessToken = token.getAccessToken();
  
  // 添加认证头
  const headers: HeadersInit = {
    'Content-Type': 'application/json',
    ...options.headers,
  };
  
  if (accessToken) {
    headers['Authorization'] = `Bearer ${accessToken}`;
  }
  
  // 发送请求
  let response = await fetch(url, { ...options, headers });
  
  // 401 时尝试刷新 Token
  if (response.status === 401 && accessToken) {
    const newToken = await refreshTokenIfNeeded();
    if (newToken) {
      headers['Authorization'] = `Bearer ${newToken}`;
      response = await fetch(url, { ...options, headers });
    } else {
      // 跳转到登录页
      window.location.href = '/login';
      throw new Error('登录已过期，请重新登录');
    }
  }
  
  return response.json();
}

// 导出请求方法
export const request = {
  get: <T>(endpoint: string, params?: object) => 
    fetchWithAuth<T>(`${endpoint}${buildQueryString(params)}`, { method: 'GET' }),
  
  post: <T>(endpoint: string, data?: unknown) =>
    fetchWithAuth<T>(endpoint, {
      method: 'POST',
      body: JSON.stringify(data),
    }),
  
  put: <T>(endpoint: string, data?: unknown) =>
    fetchWithAuth<T>(endpoint, {
      method: 'PUT',
      body: JSON.stringify(data),
    }),
  
  delete: <T>(endpoint: string, data?: unknown) =>
    fetchWithAuth<T>(endpoint, {
      method: 'DELETE',
      body: JSON.stringify(data),
    }),
};
```

### 4. 状态管理

**认证上下文**

```typescript
// src/contexts/auth-context.tsx
interface AuthState {
  user: User | null;
  loading: boolean;
  isAuthenticated: boolean;
}

interface AuthContextValue extends AuthState {
  login: (username: string, password: string) => Promise<User>;
  logout: () => void;
  register: (data: RegisterData) => Promise<void>;
  updateUser: (user: User) => void;
}

export function AuthProvider({ children }) {
  const [state, dispatch] = useReducer(authReducer, initialState);
  
  // 初始化认证状态
  useEffect(() => {
    const initAuth = async () => {
      const accessToken = token.getAccessToken();
      if (accessToken && !token.isExpired(accessToken)) {
        const res = await authService.getCurrentUser();
        dispatch({ type: 'LOGIN_SUCCESS', payload: res.data });
      }
    };
    initAuth();
  }, []);
  
  // 登录方法
  const login = useCallback(async (username, password) => {
    const res = await authService.login({ usernameOrEmail: username, password });
    token.setTokens(res.data.accessToken, res.data.refreshToken);
    dispatch({ type: 'LOGIN_SUCCESS', payload: res.data.user });
    return res.data.user;
  }, []);
  
  return (
    <AuthContext.Provider value={{ ...state, login, logout, register, updateUser }}>
      {children}
    </AuthContext.Provider>
  );
}
```

### 认证流程

**登录流程**

```
用户输入 → 表单验证 → 调用 authService.login()
                              ↓
                    发送 POST /auth/login
                              ↓
                    ┌─────────┴─────────┐
                    │                   │
              Mock 模式            真实后端模式
                    │                   │
            MSW 拦截请求          发送到后端服务器
                    │                   │
            查询 mockUsers        查询数据库
                    │                   │
            生成 Mock Token       生成 JWT Token
                    │                   │
                    └─────────┬─────────┘
                              ↓
                    返回 { user, accessToken, refreshToken }
                              ↓
                    保存到 LocalStorage
                              ↓
                    更新 AuthContext
                              ↓
                    跳转到首页
```

**Token 刷新流程**

```
API 请求 → 检查 Token 是否过期
                ↓
            Token 有效 → 直接发送请求
                ↓
            Token 过期 → 调用刷新接口
                              ↓
                    POST /auth/refresh
                              ↓
                    返回新的 Token
                              ↓
                    更新 LocalStorage
                              ↓
                    重新发送原请求
```

### 路由守卫

**认证守卫**

```typescript
// 需要登录的页面
export default function ProtectedPage() {
  const { isAuthenticated, loading } = useAuth();
  const router = useRouter();
  
  useEffect(() => {
    if (!loading && !isAuthenticated) {
      router.push('/login');
    }
  }, [isAuthenticated, loading, router]);
  
  if (loading) return <Loading />;
  if (!isAuthenticated) return null;
  
  return <PageContent />;
}
```

**角色守卫**

```typescript
// 需要管理员权限的页面
export default function AdminPage() {
  const { user, isAuthenticated } = useAuth();
  const router = useRouter();
  
  useEffect(() => {
    if (isAuthenticated && user?.role !== 'ADMIN') {
      router.push('/');
    }
  }, [user, isAuthenticated, router]);
  
  if (user?.role !== 'ADMIN') return null;
  
  return <AdminContent />;
}
```
