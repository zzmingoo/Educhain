## 2. 数据库优化

### 2.1 索引优化

```python
# 创建索引
CREATE INDEX idx_blocks_index ON blocks(index);
CREATE INDEX idx_blocks_hash ON blocks(hash);
CREATE INDEX idx_blocks_timestamp ON blocks(timestamp);

# 复合索引
CREATE INDEX idx_blocks_index_timestamp ON blocks(index, timestamp);
```

**索引策略**：
- 为频繁查询的字段创建索引
- 使用复合索引优化多条件查询
- 避免过多索引影响写入性能
- 定期分析索引使用情况

```sql
-- 分析索引使用（PostgreSQL）
SELECT schemaname, tablename, indexname, idx_scan
FROM pg_stat_user_indexes
ORDER BY idx_scan;
```

### 3.2 缓存策略

```python
from functools import lru_cache
from cachetools import TTLCache
import time

# 内存缓存（LRU）
@lru_cache(maxsize=1000)
def get_block_by_index(index: int) -> Optional[Block]:
    """缓存区块查询"""
    return blockchain.get_block_by_index(index)

# 带过期时间的缓存
cache = TTLCache(maxsize=1000, ttl=300)  # 5分钟过期

def get_chain_info_cached():
    """缓存区块链信息"""
    if 'chain_info' not in cache:
        cache['chain_info'] = blockchain.get_chain_info()
    return cache['chain_info']

# Redis 缓存
import redis
redis_client = redis.Redis(host='localhost', port=6379, db=0)

def get_transaction_cached(knowledge_id: int):
    """使用Redis缓存交易"""
    cache_key = f"tx:{knowledge_id}"
    
    # 尝试从缓存获取
    cached = redis_client.get(cache_key)
    if cached:
        return json.loads(cached)
    
    # 从数据库查询
    transaction = blockchain.get_transaction_by_knowledge_id(knowledge_id)
    
    # 存入缓存（1小时过期）
    if transaction:
        redis_client.setex(
            cache_key,
            3600,
            json.dumps(transaction.to_dict())
        )
    
    return transaction
```

### 3.3 批量操作

```python
def certify_batch(requests: List[CertifyRequest]) -> List[TransactionResponse]:
    """批量存证"""
    responses = []
    
    # 批量创建交易
    transactions = []
    for req in requests:
        tx = Transaction(
            type=req.type,
            knowledge_id=req.knowledge_id,
            user_id=req.user_id,
            content_hash=req.content_hash
        )
        transactions.append(tx)
        blockchain.add_transaction(tx)
    
    # 一次性创建区块
    block = blockchain.create_block()
    
    # 批量保存到数据库
    db_manager.save_block(block)
    
    # 生成响应
    for i, tx in enumerate(transactions):
        responses.append(TransactionResponse(
            transaction_id=i,
            block_index=block.index,
            status="confirmed",
            timestamp=tx.timestamp
        ))
    
    return responses
```

```python
        """缓存二维码生成"""
```

## 6. 监控和分析

### 6.1 性能监控

```python
import time
from fastapi import Request

@app.middleware("http")
async def add_process_time_header(request: Request, call_next):
    """记录请求处理时间"""
    start_time = time.time()
    response = await call_next(request)
    process_time = time.time() - start_time
    
    response.headers["X-Process-Time"] = str(process_time)
    
    # 记录慢请求
    if process_time > 1.0:
        logging.warning(
            f"Slow request: {request.method} {request.url} "
            f"took {process_time:.2f}s"
        )
    
    return response
```

### 6.2 Prometheus 监控

```python
from prometheus_client import Counter, Histogram, generate_latest
from fastapi import Response

# 定义指标
request_count = Counter(
    'http_requests_total',
    'Total HTTP requests',
    ['method', 'endpoint', 'status']
)

request_duration = Histogram(
    'http_request_duration_seconds',
    'HTTP request duration',
    ['method', 'endpoint']
)

@app.get("/metrics")
async def metrics():
    """Prometheus 指标端点"""
    return Response(
        content=generate_latest(),
        media_type="text/plain"
    )

@app.middleware("http")
async def prometheus_middleware(request: Request, call_next):
    """记录 Prometheus 指标"""
    start_time = time.time()
    response = await call_next(request)
    duration = time.time() - start_time
    
    request_count.labels(
        method=request.method,
        endpoint=request.url.path,
        status=response.status_code
    ).inc()
    
    request_duration.labels(
        method=request.method,
        endpoint=request.url.path
    ).observe(duration)
    
    return response
```

### 6.3 性能分析

```python
import cProfile
import pstats

def profile_function(func):
    """性能分析装饰器"""
    def wrapper(*args, **kwargs):
        profiler = cProfile.Profile()
        profiler.enable()
        
        result = func(*args, **kwargs)
        
        profiler.disable()
        stats = pstats.Stats(profiler)
        stats.sort_stats('cumulative')
        stats.print_stats(10)  # 打印前10个最慢的函数
        
        return result
    return wrapper
```
