```python

    pass
```

### 2.3 类型注解

```python
from typing import List, Optional, Dict, Any

def add_transaction(
    self, 
    transaction: Transaction
) -> int:
    """添加交易到待处理队列
    
    Args:
        transaction: 交易对象
    
    Returns:
        交易将被添加到的区块索引
    
    Raises:
        ValueError: 如果交易无效
    """
    if not transaction.validate():
        raise ValueError("Invalid transaction")
    
    self.pending_transactions.append(transaction)
    return self.get_latest_block().index + 1
```

### 2.4 文档字符串

使用 **Google 风格** 的文档字符串：

```python
def verify_knowledge(self, knowledge_id: int, content_hash: str) -> bool:
    """验证知识内容
    
    通过比对区块链上的内容哈希值，验证内容的真实性和完整性。
    
    Args:
        knowledge_id: 知识内容ID
        content_hash: 内容SHA-256哈希值
    
    Returns:
        验证是否通过，True表示内容未被篡改
    
    Examples:
        >>> blockchain.verify_knowledge(123, "a1b2c3...")
        True
    """
    transaction = self.get_transaction_by_knowledge_id(knowledge_id)
    if not transaction:
        return False
    
    return transaction.content_hash == content_hash
```

## 3. 测试

### 3.1 单元测试

创建测试文件 `tests/test_blockchain.py`:

```python
import pytest
from app.blockchain import Blockchain
from app.transaction import Transaction

class TestBlockchain:
    def setup_method(self):
        """每个测试方法前执行"""
        self.blockchain = Blockchain()
    
    def test_create_genesis_block(self):
        """测试创世区块创建"""
        assert len(self.blockchain.chain) == 1
        assert self.blockchain.chain[0].index == 0
        assert self.blockchain.chain[0].previous_hash == '0' * 64
    
    def test_add_transaction(self):
        """测试添加交易"""
        transaction = Transaction(
            type=Transaction.TYPE_KNOWLEDGE_CERT,
            knowledge_id=123,
            user_id=456,
            content_hash="a" * 64
        )
        
        block_index = self.blockchain.add_transaction(transaction)
        assert block_index == 1
        assert len(self.blockchain.pending_transactions) == 1
    
    def test_create_block(self):
        """测试创建区块"""
        transaction = Transaction(
            type=Transaction.TYPE_KNOWLEDGE_CERT,
            knowledge_id=123,
            user_id=456,
            content_hash="a" * 64
        )
        
        self.blockchain.add_transaction(transaction)
        block = self.blockchain.create_block()
        
        assert block is not None
        assert block.index == 1
        assert len(block.transactions) == 1
        assert len(self.blockchain.pending_transactions) == 0
    
    def test_chain_validation(self):
        """测试区块链验证"""
        # 添加几个区块
        for i in range(3):
            transaction = Transaction(
                type=Transaction.TYPE_KNOWLEDGE_CERT,
                knowledge_id=i,
                user_id=i,
                content_hash="a" * 64
            )
            self.blockchain.add_transaction(transaction)
            self.blockchain.create_block()
        
        assert self.blockchain.is_chain_valid() is True
        
        # 篡改区块
        self.blockchain.chain[1].transactions[0].content_hash = "b" * 64
        assert self.blockchain.is_chain_valid() is False
```

运行测试：

```bash
# 运行所有测试
pytest

# 运行特定文件
pytest tests/test_blockchain.py

# 运行特定测试
pytest tests/test_blockchain.py::TestBlockchain::test_create_block

# 显示详细输出
pytest -v

# 生成覆盖率报告
pytest --cov=app --cov-report=html
```

```python
from app.api import app

client = TestClient(app)
```

### 3.3 性能测试

```python
import time
import pytest

def test_certify_performance():
    """测试存证性能"""
    start_time = time.time()
    
    for i in range(100):
        response = client.post(
            "/api/blockchain/certify",
            json={
                "type": "KNOWLEDGE_CERTIFICATION",
                "knowledge_id": i,
                "user_id": i,
                "content_hash": "a" * 64
            }
        )
        assert response.status_code == 200
    
    elapsed_time = time.time() - start_time
    avg_time = elapsed_time / 100
    
    print(f"Average time per request: {avg_time:.3f}s")
    assert avg_time < 0.1  # 平均响应时间应小于100ms
```

## 4. 调试

### 4.1 日志调试

```python
import logging

# 配置日志
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)

logger = logging.getLogger(__name__)

def add_transaction(self, transaction: Transaction) -> int:
    logger.debug(f"Adding transaction: {transaction.generate_id()}")
    
    if not transaction.validate():
        logger.error(f"Invalid transaction: {transaction}")
        raise ValueError("Invalid transaction")
    
    self.pending_transactions.append(transaction)
    logger.info(f"Transaction added successfully")
    
    return self.get_latest_block().index + 1
```
